# I2C Communication - ESP32 + ESP-IDF

## Overview
This skill covers I2C (Inter-Integrated Circuit) communication on ESP32 using ESP-IDF framework. I2C is a two-wire serial protocol used for communicating with sensors and peripherals like MPU6050, DS3231 RTC, BME280, and LCD displays.

## Target Platform
- **MCU:** ESP32 (ESP32-S3, ESP32-C3, etc.)
- **Framework:** ESP-IDF
- **API:** I2C Master driver
- **Pins:** 2 pins (SDA - data, SCL - clock)
- **Max Speed:** Up to 400 kHz (Fast Mode), 1 MHz (Fast Mode Plus)

## Key Concepts
- **Master-Slave Architecture:** ESP32 is master, sensors are slaves
- **7-bit Addressing:** Devices identified by 7-bit addresses (0x00-0x7F)
- **SDA (Serial Data):** Bidirectional data line
- **SCL (Serial Clock):** Clock signal generated by master
- **Pull-up Resistors:** Required on both SDA and SCL (typically 4.7kΩ)
- **Acknowledge (ACK/NACK):** Slave acknowledges successful byte reception

## I2C Transaction Sequence
```
1. START condition
2. Send slave address + R/W bit
3. Wait for ACK
4. Send/receive data bytes
5. Wait for ACK after each byte
6. STOP condition
```

## Implementation Pattern

### Basic I2C Setup
```c
#include "driver/i2c.h"
#include "esp_log.h"

#define I2C_MASTER_SCL_IO    39        // GPIO for SCL
#define I2C_MASTER_SDA_IO    38        // GPIO for SDA
#define I2C_MASTER_NUM       I2C_NUM_0 // I2C port number
#define I2C_MASTER_FREQ_HZ   100000    // 100 kHz
#define I2C_MASTER_TIMEOUT_MS 1000

static const char *TAG = "I2C";

esp_err_t i2c_master_init(void)
{
    i2c_config_t conf = {
        .mode = I2C_MODE_MASTER,
        .sda_io_num = I2C_MASTER_SDA_IO,
        .scl_io_num = I2C_MASTER_SCL_IO,
        .sda_pullup_en = GPIO_PULLUP_ENABLE,
        .scl_pullup_en = GPIO_PULLUP_ENABLE,
        .master.clk_speed = I2C_MASTER_FREQ_HZ,
    };

    esp_err_t err = i2c_param_config(I2C_MASTER_NUM, &conf);
    if (err != ESP_OK) {
        return err;
    }

    return i2c_driver_install(I2C_MASTER_NUM, conf.mode, 0, 0, 0);
}

// Write data to I2C device
esp_err_t i2c_write_bytes(uint8_t device_addr, uint8_t reg_addr, uint8_t *data, size_t len)
{
    i2c_cmd_handle_t cmd = i2c_cmd_link_create();
    i2c_master_start(cmd);
    i2c_master_write_byte(cmd, (device_addr << 1) | I2C_MASTER_WRITE, true);
    i2c_master_write_byte(cmd, reg_addr, true);
    i2c_master_write(cmd, data, len, true);
    i2c_master_stop(cmd);

    esp_err_t ret = i2c_master_cmd_begin(I2C_MASTER_NUM, cmd, pdMS_TO_TICKS(I2C_MASTER_TIMEOUT_MS));
    i2c_cmd_link_delete(cmd);

    return ret;
}

// Read data from I2C device
esp_err_t i2c_read_bytes(uint8_t device_addr, uint8_t reg_addr, uint8_t *data, size_t len)
{
    i2c_cmd_handle_t cmd = i2c_cmd_link_create();

    // Write register address
    i2c_master_start(cmd);
    i2c_master_write_byte(cmd, (device_addr << 1) | I2C_MASTER_WRITE, true);
    i2c_master_write_byte(cmd, reg_addr, true);

    // Read data
    i2c_master_start(cmd);  // Repeated START
    i2c_master_write_byte(cmd, (device_addr << 1) | I2C_MASTER_READ, true);

    if (len > 1) {
        i2c_master_read(cmd, data, len - 1, I2C_MASTER_ACK);
    }
    i2c_master_read_byte(cmd, data + len - 1, I2C_MASTER_NACK);

    i2c_master_stop(cmd);

    esp_err_t ret = i2c_master_cmd_begin(I2C_MASTER_NUM, cmd, pdMS_TO_TICKS(I2C_MASTER_TIMEOUT_MS));
    i2c_cmd_link_delete(cmd);

    return ret;
}
```

## Complete Examples

### Example 1: MPU6050 IMU Sensor
```c
#include <stdio.h>
#include "driver/i2c.h"
#include "esp_log.h"
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"

#define MPU6050_ADDR         0x68
#define MPU6050_WHO_AM_I     0x75
#define MPU6050_PWR_MGMT_1   0x6B
#define MPU6050_ACCEL_XOUT_H 0x3B
#define MPU6050_GYRO_XOUT_H  0x43

typedef struct {
    int16_t accel_x, accel_y, accel_z;
    int16_t gyro_x, gyro_y, gyro_z;
    int16_t temperature;
} mpu6050_data_t;

esp_err_t mpu6050_init(void)
{
    uint8_t who_am_i;
    esp_err_t ret;

    // Check WHO_AM_I register
    ret = i2c_read_bytes(MPU6050_ADDR, MPU6050_WHO_AM_I, &who_am_i, 1);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Failed to read WHO_AM_I");
        return ret;
    }

    if (who_am_i != 0x68) {
        ESP_LOGE(TAG, "MPU6050 not found. WHO_AM_I: 0x%02X", who_am_i);
        return ESP_FAIL;
    }

    // Wake up MPU6050 (clear sleep bit)
    uint8_t pwr_mgmt = 0x00;
    ret = i2c_write_bytes(MPU6050_ADDR, MPU6050_PWR_MGMT_1, &pwr_mgmt, 1);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Failed to wake up MPU6050");
        return ret;
    }

    vTaskDelay(pdMS_TO_TICKS(100));  // Wait for sensor to stabilize

    ESP_LOGI(TAG, "MPU6050 initialized successfully");
    return ESP_OK;
}

esp_err_t mpu6050_read_data(mpu6050_data_t *data)
{
    uint8_t buffer[14];
    esp_err_t ret;

    // Read 14 bytes starting from ACCEL_XOUT_H
    ret = i2c_read_bytes(MPU6050_ADDR, MPU6050_ACCEL_XOUT_H, buffer, 14);
    if (ret != ESP_OK) {
        return ret;
    }

    // Parse data (big-endian)
    data->accel_x = (buffer[0] << 8) | buffer[1];
    data->accel_y = (buffer[2] << 8) | buffer[3];
    data->accel_z = (buffer[4] << 8) | buffer[5];
    data->temperature = (buffer[6] << 8) | buffer[7];
    data->gyro_x = (buffer[8] << 8) | buffer[9];
    data->gyro_y = (buffer[10] << 8) | buffer[11];
    data->gyro_z = (buffer[12] << 8) | buffer[13];

    return ESP_OK;
}

void app_main(void)
{
    ESP_ERROR_CHECK(i2c_master_init());
    ESP_ERROR_CHECK(mpu6050_init());

    mpu6050_data_t imu_data;

    while (1) {
        if (mpu6050_read_data(&imu_data) == ESP_OK) {
            // Convert to g and deg/s
            float accel_x_g = imu_data.accel_x / 16384.0;
            float accel_y_g = imu_data.accel_y / 16384.0;
            float accel_z_g = imu_data.accel_z / 16384.0;

            float gyro_x_dps = imu_data.gyro_x / 131.0;
            float gyro_y_dps = imu_data.gyro_y / 131.0;
            float gyro_z_dps = imu_data.gyro_z / 131.0;

            printf("Accel: %.2f, %.2f, %.2f g | Gyro: %.2f, %.2f, %.2f dps\n",
                   accel_x_g, accel_y_g, accel_z_g,
                   gyro_x_dps, gyro_y_dps, gyro_z_dps);
        }

        vTaskDelay(pdMS_TO_TICKS(100));
    }
}
```

### Example 2: DS3231 RTC (Real-Time Clock)
```c
#define DS3231_ADDR     0x68
#define DS3231_TIME_REG 0x00
#define DS3231_TEMP_REG 0x11

typedef struct {
    uint8_t second;
    uint8_t minute;
    uint8_t hour;
    uint8_t day;
    uint8_t date;
    uint8_t month;
    uint8_t year;
} rtc_time_t;

// BCD to decimal conversion
uint8_t bcd_to_dec(uint8_t bcd) {
    return ((bcd >> 4) * 10) + (bcd & 0x0F);
}

// Decimal to BCD conversion
uint8_t dec_to_bcd(uint8_t dec) {
    return ((dec / 10) << 4) | (dec % 10);
}

esp_err_t ds3231_set_time(rtc_time_t *time)
{
    uint8_t data[7];

    data[0] = dec_to_bcd(time->second);
    data[1] = dec_to_bcd(time->minute);
    data[2] = dec_to_bcd(time->hour);
    data[3] = dec_to_bcd(time->day);
    data[4] = dec_to_bcd(time->date);
    data[5] = dec_to_bcd(time->month);
    data[6] = dec_to_bcd(time->year);

    return i2c_write_bytes(DS3231_ADDR, DS3231_TIME_REG, data, 7);
}

esp_err_t ds3231_get_time(rtc_time_t *time)
{
    uint8_t data[7];
    esp_err_t ret;

    ret = i2c_read_bytes(DS3231_ADDR, DS3231_TIME_REG, data, 7);
    if (ret != ESP_OK) {
        return ret;
    }

    time->second = bcd_to_dec(data[0] & 0x7F);
    time->minute = bcd_to_dec(data[1] & 0x7F);
    time->hour = bcd_to_dec(data[2] & 0x3F);
    time->day = bcd_to_dec(data[3] & 0x07);
    time->date = bcd_to_dec(data[4] & 0x3F);
    time->month = bcd_to_dec(data[5] & 0x1F);
    time->year = bcd_to_dec(data[6]);

    return ESP_OK;
}

float ds3231_get_temperature(void)
{
    uint8_t data[2];
    esp_err_t ret;

    ret = i2c_read_bytes(DS3231_ADDR, DS3231_TEMP_REG, data, 2);
    if (ret != ESP_OK) {
        return -999.0;
    }

    // Temperature = MSB + (MSB >> 6) * 0.25
    int8_t temp_msb = data[0];
    uint8_t temp_lsb = data[1];

    float temperature = temp_msb + ((temp_lsb >> 6) * 0.25);

    return temperature;
}

void app_main(void)
{
    ESP_ERROR_CHECK(i2c_master_init());

    // Set time to 2026/02/02 15:37:00
    rtc_time_t set_time = {
        .second = 0,
        .minute = 37,
        .hour = 15,
        .day = 1,      // Sunday = 1
        .date = 2,
        .month = 2,
        .year = 26     // 2026
    };

    ESP_ERROR_CHECK(ds3231_set_time(&set_time));
    ESP_LOGI(TAG, "Time set to 2026/02/02 15:37:00");

    rtc_time_t current_time;

    while (1) {
        if (ds3231_get_time(&current_time) == ESP_OK) {
            printf("Date: 20%02d/%02d/%02d Time: %02d:%02d:%02d\n",
                   current_time.year, current_time.month, current_time.date,
                   current_time.hour, current_time.minute, current_time.second);

            float temp = ds3231_get_temperature();
            printf("Temperature: %.2f °C\n", temp);
        }

        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}
```

### Example 3: BME280 Environmental Sensor
```c
#define BME280_ADDR         0x76  // or 0x77
#define BME280_CHIP_ID_REG  0xD0
#define BME280_RESET_REG    0xE0
#define BME280_CTRL_MEAS    0xF4
#define BME280_CONFIG       0xF5
#define BME280_PRESS_MSB    0xF7

esp_err_t bme280_init(void)
{
    uint8_t chip_id;
    esp_err_t ret;

    // Read chip ID
    ret = i2c_read_bytes(BME280_ADDR, BME280_CHIP_ID_REG, &chip_id, 1);
    if (ret != ESP_OK) {
        ESP_LOGE(TAG, "Failed to read BME280 chip ID");
        return ret;
    }

    if (chip_id != 0x60) {
        ESP_LOGE(TAG, "BME280 not found. Chip ID: 0x%02X", chip_id);
        return ESP_FAIL;
    }

    // Configure sensor
    // Normal mode, oversampling x1 for all measurements
    uint8_t ctrl_meas = 0x27;  // osrs_t=1, osrs_p=1, mode=normal
    ret = i2c_write_bytes(BME280_ADDR, BME280_CTRL_MEAS, &ctrl_meas, 1);
    if (ret != ESP_OK) {
        return ret;
    }

    uint8_t config = 0xA0;  // t_sb=1000ms, filter=off
    ret = i2c_write_bytes(BME280_ADDR, BME280_CONFIG, &config, 1);
    if (ret != ESP_OK) {
        return ret;
    }

    ESP_LOGI(TAG, "BME280 initialized successfully");
    return ESP_OK;
}

esp_err_t bme280_read_raw(int32_t *temp, int32_t *press, int32_t *humid)
{
    uint8_t data[8];
    esp_err_t ret;

    ret = i2c_read_bytes(BME280_ADDR, BME280_PRESS_MSB, data, 8);
    if (ret != ESP_OK) {
        return ret;
    }

    *press = (data[0] << 12) | (data[1] << 4) | (data[2] >> 4);
    *temp = (data[3] << 12) | (data[4] << 4) | (data[5] >> 4);
    *humid = (data[6] << 8) | data[7];

    return ESP_OK;
}
```

### Example 4: I2C Device Scanner
```c
void i2c_scan(void)
{
    ESP_LOGI(TAG, "Scanning I2C bus...");

    int devices_found = 0;

    for (uint8_t addr = 1; addr < 127; addr++) {
        i2c_cmd_handle_t cmd = i2c_cmd_link_create();
        i2c_master_start(cmd);
        i2c_master_write_byte(cmd, (addr << 1) | I2C_MASTER_WRITE, true);
        i2c_master_stop(cmd);

        esp_err_t ret = i2c_master_cmd_begin(I2C_MASTER_NUM, cmd, pdMS_TO_TICKS(50));
        i2c_cmd_link_delete(cmd);

        if (ret == ESP_OK) {
            ESP_LOGI(TAG, "Device found at address 0x%02X", addr);
            devices_found++;
        }
    }

    if (devices_found == 0) {
        ESP_LOGW(TAG, "No I2C devices found");
    } else {
        ESP_LOGI(TAG, "Found %d device(s)", devices_found);
    }
}
```

## Multi-Byte Read/Write Optimization

### Burst Read (More Efficient)
```c
// Read multiple registers in one transaction
esp_err_t i2c_burst_read(uint8_t device_addr, uint8_t start_reg, uint8_t *data, size_t len)
{
    return i2c_read_bytes(device_addr, start_reg, data, len);
}
```

### Single Register Operations
```c
esp_err_t i2c_write_register(uint8_t device_addr, uint8_t reg_addr, uint8_t value)
{
    return i2c_write_bytes(device_addr, reg_addr, &value, 1);
}

esp_err_t i2c_read_register(uint8_t device_addr, uint8_t reg_addr, uint8_t *value)
{
    return i2c_read_bytes(device_addr, reg_addr, value, 1);
}

// Read-Modify-Write
esp_err_t i2c_modify_register(uint8_t device_addr, uint8_t reg_addr, uint8_t mask, uint8_t value)
{
    uint8_t reg_value;
    esp_err_t ret;

    // Read current value
    ret = i2c_read_register(device_addr, reg_addr, &reg_value);
    if (ret != ESP_OK) {
        return ret;
    }

    // Modify
    reg_value = (reg_value & ~mask) | (value & mask);

    // Write back
    return i2c_write_register(device_addr, reg_addr, reg_value);
}
```

## Error Handling and Retry Logic

```c
#define MAX_RETRIES 3

esp_err_t i2c_read_with_retry(uint8_t device_addr, uint8_t reg_addr, uint8_t *data, size_t len)
{
    esp_err_t ret;

    for (int retry = 0; retry < MAX_RETRIES; retry++) {
        ret = i2c_read_bytes(device_addr, reg_addr, data, len);

        if (ret == ESP_OK) {
            return ESP_OK;
        }

        ESP_LOGW(TAG, "I2C read failed, retry %d/%d", retry + 1, MAX_RETRIES);
        vTaskDelay(pdMS_TO_TICKS(10));
    }

    ESP_LOGE(TAG, "I2C read failed after %d retries", MAX_RETRIES);
    return ret;
}
```

## Best Practices
1. **Pull-up resistors** - Always use appropriate pull-ups (4.7kΩ typical)
2. **Bus speed** - Start with 100 kHz, increase if needed
3. **Check ACK** - Always check for acknowledgment
4. **Error handling** - Implement retry logic for reliability
5. **Device scanner** - Use scanner to verify device addresses
6. **Wire length** - Keep I2C wires short (< 1m) for reliability

## Common Pitfalls
- ❌ Missing pull-up resistors (communication fails)
- ❌ Wrong device address (no ACK received)
- ❌ Clock speed too high (unreliable communication)
- ❌ Long wires without proper termination (signal integrity issues)
- ❌ Multiple devices with same address (bus conflict)
- ❌ Not checking return values (silent failures)

## I2C Clock Speed Selection

```c
// Standard mode (100 kHz) - Most compatible
.master.clk_speed = 100000,

// Fast mode (400 kHz) - Good for most sensors
.master.clk_speed = 400000,

// Fast mode plus (1 MHz) - Only if device supports it
.master.clk_speed = 1000000,
```
